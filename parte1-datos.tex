\begin{equation}
    \label{ec:1-teoria:ganancia-experimental}
    A = \frac{v_o}{v_i}
\end{equation}

\begin{equation}
    \label{ec:1-teoria:err-ganancia-experimental}
    \Delta A = \left| \frac{1}{v_i} \right| \Delta v_o + \left| -\frac{v_o}{{v_i}^2} \right| \Delta v_i
\end{equation}
